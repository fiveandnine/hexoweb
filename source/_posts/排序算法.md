### 快速排序
```js
function quickSort(arr){
  for (var i = 0; i < arr.length; i++){
    var minKey = i
    for (var j = i+1; j <arr.length; j++){
      if(arr[i]>arr[j]){
        minKey = j
      }
    }
    var temp = arr[i];
    arr[i] = arr[minKey];
    arr[minKey] = temp;
  } 
  return arr
}
```

### 单向冒泡排序
```js
function bubbleSort(arr) {
  for (var j = 0; j< arr.length; j++)
   for (var i = 0; i < arr.length-1-j; i ++){
     if(arr[i] > arr[i+1]){
       var temp = arr[i]
       arr[i] = arr[i+1]
       arr[i+1] = temp
     }
   } 
  }
  return arr
}
```

### 双向冒泡排序

function bubbleSort_twoWay(arr){
  for (var i = 0; i< arr.length; i++){
    
    for(var j = 0; j< arr.length - 1 - i; j ++){
      if(arr[j]>arr[j+1]){
        var temp = arr[j];
        arr[j] = arr[j+1]
        arr[j+1] = temp
      }
    }
    
    for(var j = arr.length - i; j > 0; j--){

        var temp = arr[j];
        arr[j] = arr[j-1]
        arr[j-1] = temp
      }
    }
    
  } 
  return arr
}

function insertionSort1(arr) {
  for (var i = 1; i < arr.length+1; i++){
    for(var j = i-1; j >=0; j--){
      if(arr[j]> arr[j+1]){
        var temp = arr[j+1]
        arr[j+1] = arr[j];
        arr[j] = temp
      }else{
        break
      }
    }
  }

function insertionSort3(arr){
  for(var i=0; i< arr.length; i++){
    var current = arr[i]
    var preKey = i-1
    while(preKey>=0 && arr[preKey] > current){
      arr[preKey+1] = arr[preKey]
      preKey--
    }
    arr[preKey] = current
  }
}

### 希尔排序
```js
function shellSort(arr) {
  let len = arr.length;
  for (let gap = Math.floor(len / 2); gap > 0; gap = Math.floor(gap / 2)) {
    for (let i = gap; i < len; i++) {
      let j = i;
      let current = arr[i];
      while(j - gap >= 0 && current < arr[j - gap]) {
        arr[j] = arr[j - gap];
        j = j - gap;
      }
      arr[j] = current;
    }
  }
}
```
